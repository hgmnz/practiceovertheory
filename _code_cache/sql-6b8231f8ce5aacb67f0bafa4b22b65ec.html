select pg_locks.granted,
       pg_stat_activity.query,
       pg_stat_activity.query_start
       from pg_locks
  JOIN pg_stat_activity
    on pg_locks.pid = pg_stat_activity.pid
  WHERE pg_locks.locktype = 'advisory';
